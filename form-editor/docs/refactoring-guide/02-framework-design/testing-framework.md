      queries: { retry: false },

    jest.clearAllMocks();
  });

    const mockForms = [

      forms: mockForms,
      total: 2,
    });

      wrapper: createWrapper(),

      expect(result.current.isSuccess).toBe(true);

    const { result } = renderHook(() => useForms(), {
      wrapper: createWrapper(),
    });

    await waitFor(() => {
      expect(result.current.isError).toBe(true);
