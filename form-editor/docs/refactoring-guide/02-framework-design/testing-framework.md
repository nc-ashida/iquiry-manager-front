    jest.clearAllMocks();
  });

    const mockForms = [
      { id: '1', name: 'Test Form 1', fields: [] },

      total: 2,

      wrapper: createWrapper(),

      wrapper: createWrapper(),
    });

      expect(result.current.isError).toBe(true);
